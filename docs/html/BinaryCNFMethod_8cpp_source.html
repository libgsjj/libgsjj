104  void BinaryCNFMethod::createFormula() {
105  Gate *formula = nullptr;
106 
107  // For every ua, va in prefixes such that the last letter of ua is the same as the last letter of va
108  for (const auto &ua : m_prefixes) {
109  if (ua.size() > 0) {
110  const char &a = ua.at(ua.size() - 1);
111  const std::string &u = ua.substr(0, ua.size() - 1);
112 
113  for (const auto &va : m_prefixes) {
114  if (va.size() > 0 && va.at(va.size() - 1) == a) {
115  const std::string &v = va.substr(0, va.size() - 1);
116  // We must force that x_u = x_v => x_{ua} = x_{va}
117  // In other words, we force that the transition function is correct and coherent
118  // So, we create a formula for NOT (x_u = x_v) OR x_ua = x_va
119  Gate *different = phi_different(u, v);
120  Gate *eq = equal(ua, va);
121  Gate *full = m_solver.new_OR(different, eq);
122 
123  if (formula) {
124  formula = m_solver.new_AND(formula, full);
125  }
126  else {
127  formula = full;
128  }
129  }
130  }
131  }
132  }
133 
134  // Now, we want to express that for every u in Sp and every v in Sm, x_u is different than x_v
135  // In other words, u and v can not end in the same state (since u must be accepted and v rejected)
136  for (const auto &u : m_Sp) {
137  for (const auto &v : m_Sm) {
138  formula = m_solver.new_AND(formula, phi_different(u, v));
139  }
140  }
141 
142  // Finally, we must make sure that the encoding is correct even if the number of states is not a power of 2
143  // http://www.graphics.stanford.edu/~seander/bithacks.html#DetermineIfPowerOf2
144  for (const auto &u : m_prefixes) {
145  for (unsigned int i = m_numberStates ; i < pow(2, m_binarySize) ; i++) {
146  formula = m_solver.new_AND(formula, phi_not_q_i(u, i));
147  }
148  }

162  Gate *x_vj = m_stateToVar.find(std::make_pair(v, j))->second;

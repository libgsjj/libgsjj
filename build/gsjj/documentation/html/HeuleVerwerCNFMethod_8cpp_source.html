180  // First, we must seek the initial state
181  unsigned int initial = 0;
182  for (unsigned int i = 0; i < m_numberStates; i++) {
183  Var v = m_xVars.find(std::make_pair("", i))->second;
184  if (model[v] == l_True) {
185  initial = i;
186  break;
187  }
188  }
189 
190  // The initial state is accepting if f_{initial} is true
191  std::unique_ptr<DFA<char>> dfa = std::make_unique<DFA<char>>(initial, model[m_fVars.find(initial)->second] == l_True);
192 
193  // Now, we create every other state (and define the accepting ones)
194  for (unsigned int q = 0; q < m_numberStates; q++) {
195  if (q != initial) {
196  dfa->addState(q, model[m_fVars.find(q)->second] == l_True);
197  }
198  }
199 
200  // Finally, we create the transitions
201  for (unsigned int p = 0; p < m_numberStates; p++) {
202  for (const char &a : m_alphabet) {
203  for (unsigned int q = 0; q < m_numberStates; q++) {
204  if (model[m_dVars.find(std::make_tuple(p, a, q))->second] == l_True) {
205  dfa->addTransition(p, a, q);

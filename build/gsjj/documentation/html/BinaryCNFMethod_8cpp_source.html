103  void BinaryCNFMethod::createFormula() {
112  Gate *formula = nullptr;
113 
114  // For every ua, va in prefixes such that the last letter of ua is the same as the last letter of va
115  for (const auto &ua : m_prefixes) {
116  if (ua.size() > 0) {
117  const char &a = ua.at(ua.size() - 1);
118  const std::string &u = ua.substr(0, ua.size() - 1);
119 
120  for (const auto &va : m_prefixes) {
121  if (va.size() > 0 && va.at(va.size() - 1) == a) {
122  const std::string &v = va.substr(0, va.size() - 1);
123  // We must force that x_u = x_v => x_{ua} = x_{va}
124  // In other words, we force that the transition function is correct and coherent
125  // So, we create a formula for NOT (x_u = x_v) OR x_ua = x_va
126  Gate *different = phi_different(u, v);
127  Gate *eq = equal(ua, va);
128  Gate *full = m_solver.new_OR(different, eq);
129 
130  if (formula) {
131  formula = m_solver.new_AND(formula, full);
132  }
133  else {
134  formula = full;
135  }
136  }
137  }
138  }
139  }
140 
141  // Now, we want to express that for every u in Sp and every v in Sm, x_u is different than x_v
142  // In other words, u and v can not end in the same state (since u must be accepted and v rejected)
143  for (const auto &u : m_Sp) {
144  for (const auto &v : m_Sm) {
145  formula = m_solver.new_AND(formula, phi_different(u, v));
